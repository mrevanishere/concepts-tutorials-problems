// Two pointer + 1
class Solution {
public:
    vector<int> twoSum(vector<int> nums, int target) {
        int n = nums.size();
        vector<int> sol;
        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                if (nums[i] + nums[j] == target) {
                    sol.push_back(i);
                    sol.push_back(j);
                    return sol;
                }
            }
        }
        return sol;
    }
};
